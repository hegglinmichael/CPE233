#The program reads three values from port 0x30 and sums them before writing the value to output 0x40.

.EQU OUTPUT_PORT = 0x40 ; output port
.CSEG
.ORG 0x01

main:
	IN		r1, 0x30
	IN		r2, 0x30
	IN		r3, 0x30

	MOV 	r4, 0x00
	ADD		r4, r1
	ADD		r4, r2
	ADD		r4, r3
	
	OUT		r4, OUTPUT_PORT
	BRN		main
